<script>
  const baseIP = "http://10.43.144.58";

  const params = new URLSearchParams(window.location.search);
  const id = params.get("id");

  if (id) {
    window.location.href = baseIP + "/strojevi/" + id + "/";
  } else {
    document.body.innerHTML = "Greška: nema ID-a";
  }
</script>
